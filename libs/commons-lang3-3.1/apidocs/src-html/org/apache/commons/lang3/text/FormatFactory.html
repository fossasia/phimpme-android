017    package org.apache.commons.lang3.text;
018    
019    import java.text.Format;
020    import java.util.Locale;
021    
022    /**
023     * Format factory.
024     * 
025     * @since 2.4
026     * @version $Id: FormatFactory.java 1088899 2011-04-05 05:31:27Z bayard $
027     */
028    public interface FormatFactory {
029    
030        /**
031         * Create or retrieve a format instance.
032         *
033         * @param name The format type name
034         * @param arguments Arguments used to create the format instance. This allows the
035         *                  <code>FormatFactory</code> to implement the "format style"
036         *                  concept from <code>java.text.MessageFormat</code>.
037         * @param locale The locale, may be null
038         * @return The format instance
039         */
040        Format getFormat(String name, String arguments, Locale locale);
041    
042    }
